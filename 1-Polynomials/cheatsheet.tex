%========== WEEK 1 ==========
\chsection{Week 1: Numbers}

\chsubsec{Number Sets}
\begin{itemize}
\item $\mathbb{N} = \{1,2,3,\ldots\}$ Natural
\item $\mathbb{Z} = \{\ldots,-1,0,1,\ldots\}$ Integers
\item $\mathbb{Q} = \frac{p}{q}$, $p,q\in\mathbb{Z}$, $q\neq 0$ Rational
\item $\mathbb{R}$ = All real numbers
\end{itemize}
$\mathbb{N} \subset \mathbb{Z} \subset \mathbb{Q} \subset \mathbb{R}$

\chsubsec{Rational vs Irrational}
\begin{itemize}
\item Rational: terminates or repeats
\item Irrational: $\sqrt{2}, \pi, e$ (non-repeating)
\end{itemize}

\chsubsec{Fraction Operations}
\begin{align*}
\frac{a}{b} + \frac{c}{d} &= \frac{ad+bc}{bd}\\
\frac{a}{b} \times \frac{c}{d} &= \frac{ac}{bd}\\
\frac{a}{b} \div \frac{c}{d} &= \frac{a}{b} \times \frac{d}{c}
\end{align*}

\chsubsec{Order of Operations}
\textbf{PEMDAS}: Parentheses, Exponents, Mult/Div, Add/Sub

\chsubsec{Inequalities}
\begin{itemize}
\item Add/subtract: sign unchanged
\item Mult/div by positive: sign unchanged
\item \textbf{Mult/div by negative: REVERSE sign!}
\end{itemize}

\chsubsec{Interval Notation}
\begin{tabular}{@{}ll@{}}
$(a,b)$ & $a < x < b$ \\
$[a,b]$ & $a \leq x \leq b$ \\
$[a,b)$ & $a \leq x < b$ \\
$(a,\infty)$ & $x > a$
\end{tabular}

\chsubsec{Absolute Value}
$|x| = \begin{cases} x & x \geq 0\\ -x & x < 0 \end{cases}$

%========== WEEK 2 ==========
\chsection{Week 2: Linear Functions}

\chsubsec{Coordinate Plane}
Point $(x,y)$: $x$ = horizontal, $y$ = vertical

\textbf{Quadrants:}\\
I: $(+,+)$, II: $(-,+)$, III: $(-,-)$, IV: $(+,-)$

\chsubsec{Functions}
$f$: each input $\to$ exactly one output

\textbf{Vertical Line Test:} Graph is a function if every vertical line hits at most once

\chsubsec{Slope}
\[\boxed{m = \frac{y_2 - y_1}{x_2 - x_1} = \frac{\text{rise}}{\text{run}}}\]
\begin{itemize}
\item $m > 0$: rises left to right
\item $m < 0$: falls left to right
\item $m = 0$: horizontal
\item $m$ undefined: vertical
\end{itemize}

\chsubsec{Linear Equations}
\begin{tabular}{@{}ll@{}}
Slope-intercept & $y = mx + b$ \\
Point-slope & $y - y_1 = m(x - x_1)$ \\
Standard & $ax + by = c$
\end{tabular}

\chsubsec{Intercepts}
\begin{itemize}
\item $y$-intercept: set $x=0$, get $(0,b)$
\item $x$-intercept: set $y=0$, solve for $x$
\end{itemize}

\chsubsec{Parallel \& Perpendicular}
\begin{itemize}
\item Parallel: $m_1 = m_2$
\item Perpendicular: $m_1 \cdot m_2 = -1$
\end{itemize}

%========== WEEK 3 ==========
\chsection{Week 3: Polynomials}

\chsubsec{Polynomial Form}
$P(x) = a_nx^n + a_{n-1}x^{n-1} + \cdots + a_0$
\begin{itemize}
\item Degree: highest power
\item Leading coeff: $a_n$
\item Constant: $a_0$
\end{itemize}

\chsubsec{Degree Names}
0: constant, 1: linear, 2: quadratic\\
3: cubic, 4: quartic, 5: quintic

\chsubsec{Special Products}
\begin{align*}
(a+b)^2 &= a^2 + 2ab + b^2\\
(a-b)^2 &= a^2 - 2ab + b^2\\
(a+b)(a-b) &= a^2 - b^2
\end{align*}

\chsubsec{Quadratic Formula}
For $ax^2 + bx + c = 0$:
\[\boxed{x = \frac{-b \pm \sqrt{b^2-4ac}}{2a}}\]

\chsubsec{Discriminant}
$\Delta = b^2 - 4ac$
\begin{itemize}
\item $\Delta > 0$: 2 real roots
\item $\Delta = 0$: 1 repeated root
\item $\Delta < 0$: no real roots
\end{itemize}

\chsubsec{Parabola Features}
$y = ax^2 + bx + c$
\begin{itemize}
\item Vertex: $x = -\frac{b}{2a}$
\item $a>0$: opens up, $a<0$: opens down
\item Axis of symmetry: $x = -\frac{b}{2a}$
\end{itemize}

\chsubsec{End Behavior}
\begin{center}
\small
\begin{tabular}{|c|c|c|}
\hline
& $a_n>0$ & $a_n<0$ \\
\hline
Even & $\uparrow\uparrow$ & $\downarrow\downarrow$ \\
Odd & $\swarrow\nearrow$ & $\nwarrow\searrow$ \\
\hline
\end{tabular}
\end{center}

\chsubsec{Roots \& Multiplicity}
\begin{itemize}
\item Degree $n$: at most $n$ roots
\item Odd mult: crosses $x$-axis
\item Even mult: bounces off $x$-axis
\end{itemize}

\chsubsec{Turning Points}
Degree $n$ $\Rightarrow$ at most $n-1$ turning points

%========== WEEK 4 ==========
\chsection{Week 4: Functions \& Graphs}

\chsubsec{Domain \& Range}
\begin{itemize}
\item Domain: all valid inputs
\item Range: all possible outputs
\end{itemize}

\textbf{Finding Domain:}
\begin{itemize}
\item Polynomials: $\mathbb{R}$ (all reals)
\item $\frac{1}{x-a}$: $x \neq a$
\item $\sqrt{x-a}$: $x \geq a$
\end{itemize}

\chsubsec{Increasing/Decreasing}
\begin{itemize}
\item Increasing: graph goes up (left to right)
\item Decreasing: graph goes down
\end{itemize}

\chsubsec{Continuity}
Continuous = no breaks, holes, or jumps\\
\textbf{All polynomials are continuous}

\textbf{Discontinuities:}
\begin{itemize}
\item Hole (removable)
\item Jump
\item Asymptote (infinite)
\end{itemize}

\chsubsec{Function Operations}
\begin{align*}
(f+g)(x) &= f(x) + g(x)\\
(f \cdot g)(x) &= f(x) \cdot g(x)\\
(f/g)(x) &= f(x)/g(x)
\end{align*}

\chsubsec{Composition}
\[\boxed{(f \circ g)(x) = f(g(x))}\]
Apply $g$ first, then $f$\\
\textbf{Order matters!} $f \circ g \neq g \circ f$

\chsubsec{Transformations}
\begin{tabular}{@{}ll@{}}
$f(x)+k$ & up $k$ \\
$f(x)-k$ & down $k$ \\
$f(x-h)$ & right $h$ \\
$f(x+h)$ & left $h$ \\
$-f(x)$ & reflect $x$-axis \\
$f(-x)$ & reflect $y$-axis \\
$af(x)$ & stretch ($a>1$) \\
\end{tabular}
