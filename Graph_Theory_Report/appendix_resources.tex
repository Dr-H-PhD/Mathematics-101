\section{Useful Python Libraries}
\subsection{Graph Processing}
\begin{itemize}
	\item \textbf{NetworkX:} Comprehensive graph analysis library
	\begin{itemize}
		\item Installation: \texttt{pip install networkx}
		\item Features: Graph creation, algorithms, visualization
		\item Documentation: \url{https://networkx.org/}
	\end{itemize}
	
	\item \textbf{igraph:} High-performance graph library
	\begin{itemize}
		\item Installation: \texttt{pip install python-igraph}
		\item Features: Fast algorithms, community detection
		\item Documentation: \url{https://igraph.org/python/}
	\end{itemize}
\end{itemize}

\subsection{Graph Neural Networks}
\begin{itemize}
	\item \textbf{PyTorch Geometric:} GNN library for PyTorch
	\begin{itemize}
		\item Installation: \texttt{pip install torch-geometric}
		\item Features: GCN, GAT, GraphSAGE implementations
		\item Documentation: \url{https://pytorch-geometric.readthedocs.io/}
	\end{itemize}
	
	\item \textbf{DGL (Deep Graph Library):} Framework-agnostic GNN library
	\begin{itemize}
		\item Installation: \texttt{pip install dgl}
		\item Features: Supports PyTorch, TensorFlow, MXNet
		\item Documentation: \url{https://www.dgl.ai/}
	\end{itemize}
\end{itemize}
